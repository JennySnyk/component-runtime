078    default Entry getEntry(final String name) {
079        return Optional
080                .ofNullable(getEntries()) //
081                .orElse(Collections.emptyList()) //
082                .stream() //
083                .filter((Entry e) -> Objects.equals(e.getName(), name)) //
084                .findFirst() //
085                .orElse(null);
086    }

100     * Get a property values from schema with its name.

103     * @return property's value.

105    default JsonValue getJsonProp(final String name) {
106        final String prop = this.getProp(name);
107        if (prop == null) {
108            return null;
109        }
110        try {
111            return Json.createParser(new StringReader(prop)).getValue();
112        } catch (RuntimeException ex) {
113            return Json.createValue(prop);
114        }
115    }

117    enum Type {
118
119        RECORD(new Class<?>[] { Record.class }),
120        ARRAY(new Class<?>[] { Collection.class }),
121        STRING(new Class<?>[] { String.class }),
122        BYTES(new Class<?>[] { byte[].class, Byte[].class }),
123        INT(new Class<?>[] { Integer.class }),
124        LONG(new Class<?>[] { Long.class }),
125        FLOAT(new Class<?>[] { Float.class }),
126        DOUBLE(new Class<?>[] { Double.class }),
127        BOOLEAN(new Class<?>[] { Boolean.class }),
128        DATETIME(new Class<?>[] { Long.class, Date.class, Temporal.class });
129
130        /** All compatibles Java classes */
131        private final Class<?>[] classes;
132
133        Type(final Class<?>[] classes) {
134            this.classes = classes;
135        }
136
137        /**
138         * Check if input can be affected to an entry of this type.
139         * 
140         * @param input : object.
141         * @return true if input is null or ok.
142         */
143        public boolean isCompatible(final Object input) {
144            if (input == null) {
145                return true;
146            }
147            for (final Class<?> clazz : classes) {
148                if (clazz.isInstance(input)) {
149                    return true;
150                }
151            }
152            return false;
153        }
154    }

401    /**
402     * Sanitize name to be avro compatible.
403     * 
404     * @param name : original name.
405     * @return avro compatible name.
406     */
407    static String sanitizeConnectionName(final String name) {
408        if (name == null || name.isEmpty()) {
409            return name;
410        }
411
412        char current = name.charAt(0);
413        final CharsetEncoder ascii = Charset.forName(StandardCharsets.US_ASCII.name()).newEncoder();
414        final boolean skipFirstChar = ((!ascii.canEncode(current)) || (!Character.isLetter(current) && current != '_'))
415                && name.length() > 1 && (!Character.isDigit(name.charAt(1)));
416
417        final StringBuilder sanitizedBuilder = new StringBuilder();
418
419        if (!skipFirstChar) {
420            if (((!Character.isLetter(current)) && current != '_') || (!ascii.canEncode(current))) {
421                sanitizedBuilder.append('_');
422            } else {
423                sanitizedBuilder.append(current);
424            }
425        }
426        for (int i = 1; i < name.length(); i++) {
427            current = name.charAt(i);
428            if (!ascii.canEncode(current)) {
429                if (Character.isLowerCase(current) || Character.isUpperCase(current)) {
430                    sanitizedBuilder.append('_');
431                } else {
432                    final byte[] encoded =
433                            Base64.getEncoder().encode(name.substring(i, i + 1).getBytes(StandardCharsets.UTF_8));
434                    final String enc = new String(encoded);
435                    if (sanitizedBuilder.length() == 0 && Character.isDigit(enc.charAt(0))) {
436                        sanitizedBuilder.append('_');
437                    }
438                    for (int iter = 0; iter < enc.length(); iter++) {
439                        if (Character.isLetterOrDigit(enc.charAt(iter))) {
440                            sanitizedBuilder.append(enc.charAt(iter));
441                        } else {
442                            sanitizedBuilder.append('_');
443                        }
444                    }
445                }
446            } else if (Character.isLetterOrDigit(current)) {
447                sanitizedBuilder.append(current);
448            } else {
449                sanitizedBuilder.append('_');
450            }
451
452        }
453        return sanitizedBuilder.toString();
454    }

456    static Schema.Entry avoidCollision(final Schema.Entry newEntry,
457            final Supplier<Stream<Schema.Entry>> allEntriesSupplier, final BiConsumer<String, Entry> replaceFunction) {
458        final Optional<Entry> collisionedEntry = allEntriesSupplier //
459                .get() //
460                .filter((final Entry field) -> field.getName().equals(newEntry.getName())
461                        && !Objects.equals(field, newEntry)) //
462                .findFirst();
463        if (!collisionedEntry.isPresent()) {
464            // No collision, return new entry.
465            return newEntry;
466        }
467        final Entry matchedEntry = collisionedEntry.get();
468        final boolean matchedToChange = matchedEntry.getRawName() != null && !(matchedEntry.getRawName().isEmpty());
469        if (matchedToChange) {
470            // the rename has to be applied on entry already inside schema, so replace.
471            replaceFunction.accept(matchedEntry.getName(), newEntry);
472        } else if (newEntry.getRawName() == null || newEntry.getRawName().isEmpty()) {
473            // try to add exactly same raw, skip the add here.
474            return null;
475        }
476        final Entry fieldToChange = matchedToChange ? matchedEntry : newEntry;
477        int indexForAnticollision = 1;
478        final String baseName = Schema.sanitizeConnectionName(fieldToChange.rawName); // recalc primiti name.
479
480        String newName = baseName + "_" + indexForAnticollision;
481        final Set<String> existingNames = allEntriesSupplier //
482                .get() //
483                .map(Entry::getName) //
484                .collect(Collectors.toSet());
485        while (existingNames.contains(newName)) {
486            indexForAnticollision++;
487            newName = baseName + "_" + indexForAnticollision;
488        }
489        final Entry newFieldToAdd = fieldToChange.toBuilder().withName(newName).build();
490
491        return newFieldToAdd; // matchedToChange ? newFieldToAdd : newEntry;
492    }
